def test_ConfigPath(
    config_path: Path,
) -> None:
    """
    Perform various tests associated with relevant routines.

    :param config_path: Custom fixture for populating paths.
    """

    path = ConfigPath(config_path)


    attrs = lattrs(path)

    assert attrs == [
        'path',
        'config']


    assert inrepr(
        'paths.ConfigPath object',
        path)

    assert isinstance(
        hash(path), int)

    assert instr(
        'paths.ConfigPath object',
        path)


    assert 'test' in path.path.name

    assert len(path.config) == 1

def test_ConfigPaths(
    paths: ConfigPaths,
) -> None:
    """
    Perform various tests associated with relevant routines.

    :param paths: Custom fixture for the configuration paths.
    """


    attrs = lattrs(paths)

    assert attrs == [
        'paths', 'config',
        '_ConfigPaths__merge']


    assert inrepr(
        'paths.ConfigPaths object',
        paths)

    assert isinstance(
        hash(paths), int)

    assert instr(
        'paths.ConfigPaths object',
        paths)


    assert len(paths.paths) == 2

    assert len(paths.config) == 2


    replaces = {
        'PROJECT': PROJECT}

    sample_path = (
        SAMPLES / 'paths.json')

    sample = load_sample(
        path=sample_path,
        update=ENPYRWS,
        content=paths.merge,
        replace=replaces)

    expect = prep_sample(
        content=paths.merge,
        replace=replaces)

    assert expect == sample
